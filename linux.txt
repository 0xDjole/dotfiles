Installing custom arch linux with i3 steps:

0. DISABLE ALL SECURE FEATURES BY ANY LAPTOP OVER OR DIE :)

1. Install arch linux ISO from:
http://mirror.pmf.kg.ac.rs/archlinux/iso/2021.06.01/

2. Make ISO bootable ( put ISO file to usb )

3. Open bios on start laptop ( usually F2, with plugged bootable USB )

4. Disable secure boot in bios.

5. Put usb order first in bios ( select device and put on top of boot order )

6. iwctl
	device list
	station {device name} scan
	station {device name} get-networks
	station {device name} connect {SSID} ( Enter passphrase )
	exit
	

7. ping google.com ( Confirm net )
8. timedatectl set-ntp
9. timedatectl status
10. fdisk -l
11. fdisk /dev/{ Hardisk/SSD name }
	g
	n   ( Create boot partition, make it 550M )
		1
		Enter
		+550M
	n   ( Create swap partition, make it 2G )
        	2
        	Enter
        	+2G
	n   ( Create filesystem partition, let it take all space )
        	3
        	Enter
        	Enter
	t 
		1
		1
        t
		2
		19
	w
12. mkfs.fat -F32 /dev/{ Device name partition 1 }
13. mkswap /dev/{ Device name partition 2 }
14. swapon /dev/{ Device name partition 2 }
15. mkfs.ext4 /dev/{ Device name partition 3 }
16. mount /dev/{ Device name partition 3 } /mnt
17. pacstrap /mnt base linux linux-firmware
18. genfstab -U /mnt >> /mnt/etc/fstab
19. arch-chroot /mnt
20. ls /usr/share/zoneinfo
21. ln -sf /usr/share/zoneinfo/Europe/Belgrade /etc/localtime
22. hwclock --systohc
23. pacman -S neovim sudo grub efibootmgr dosfstools os-prober mtools networkmanager base-devel git xorg xorg-xinit nitrogen discord nautilus code flameshot alacritty i3-gaps i3blocks i3lock i3status noto-fonts ttf-font-awesome ttf-dejavu ttf-liberation nodejs npm alsa-utils bc rofi wmctrl xdotool ripgrep
24. vim ~/.bashrc && source ~/.bashrc ( check Djole .bashrc file ) 
24. vim /etc/locale.gen ( uncomment line 177 )
26. locale-gen
27. vim /etc/hostname ( Write in your hostname.., We choose "arch" )
28. vim /etc/hosts
     127.0.0.1 		localhost
     ::1 		localhost
     127.0.1.1 		{hostname}.localdomain	{hostname}
29. passwd
30. useradd -m {user name}
31. passwd {user name}
32. usermod -aG wheel,audio,video,optical,storage,input {username}
34. EDITOR=nvim visudo
35. Uncomment wheel all group
37. mkdir /boot/EFI
38. mount /dev/{ First boot partition } /boot/EFI
39. grub-install --target=x86_64-efi --bootloader-id=grub_uefi --recheck
40. grub-mkconfig -o /boot/grub/grub.cfg
41. systemctl enable NetworkManager.service
42. git clone https://aur.archlinux.org/yay-git.git && cd yay-git && makepkg -si && cd ..  && rm -rf yay-git
43. yay -S google-chrome bumblebee-status libinput-gestures
44. git clone http://github.com/creationix/nvm.git .nvm
47. source ~/.nvm/nvm.sh
48. setup nvim ( check Djole nvim file )
49. alsa-mixer
50. cp /usr/share/X11/xorg.conf.d/40-libinput.conf /etc/X11/xorg.conf.d/40-libinput.conf
51. Add to touchpad: 
	Option "NaturalScrolling" "true"
	Option "Tapping" "true"
52. vim ~/.bash_login, check djole setup
53. vim /bin/bright && chmod a+x /bin/bright
54. 
53. umount -I /mnt
